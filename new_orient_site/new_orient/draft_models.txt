class InputAudio(models.Model):
  classTag = models.CharField(max_length=20, default='inputAudio')
  productID = models.IntegerField()
  inputID = models.IntegerField() 
  physicalConnection = models.BooleanField(default='true') # true:有線, false:無線
  signalType = models.BooleanField(default='true') # true:デジタル, false:アナログ
  
  def __init__(self, prodID, numsIA):
    self.productID = prodID
    self.inputID = numsIA
  
class InputAudioWiredAnalogue(models.Model):
  physicalConProtocol = models.CharField(max_length=20)
  inputNominal = models.CharField(max_length=20) # +4dBu, -10dBV, 24dBFs
    # MIC_dBu, LINE_dBu, LINE_dBV, AMP
  connectorType = models.CharField(max_length=50) # XLR3P, XLR2P, XLR4P, RCA, EuroBlock, 6.3Phone, 3.5mini_M, 3.5mini_S
  
class InputAudioWiredDigital(models.Model):
  physicalConProtocol = models.CharField(max_length=20)
    # 有線デジタル: DANTE, COBRANET, EtherSound, AES67, AES_EBU110, AES_EBUOPT, SPDIF, YDIF
  isEther = models.BooleanField(default='false') # true: DANTE, COBRANET, EtherSound, AES67
  connectorType = models.CharField(max_length=50) # STP-4P(CAT5E), UTP-4P(CAT6), STP-4P(CAT6)
